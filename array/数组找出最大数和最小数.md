### 数组找出最大数和最小数

此版本返回max,min和最大数，最小数的索引。切片总和，平均值

max和min的值必须为切片的[0]索引，否则判断结果不准确

```go
package main

import "fmt"

func findMaxAndMin(s []int) (max, min, maxValIndex, minValIndex, sum int, avg float32) {

	max = s[0] //初始化max
	min = s[0] //初始化min

	for i := 0; i < len(s); i++ {

		//获取最大值
		if s[i] > max {
			max = s[i]
			maxValIndex = i
		}

		//获取最小值
		if s[i] < min {
			min = s[i]
			minValIndex = i
		}
		//计算总和
		sum += s[i]
	}

	avg = float32(sum / len(s))

	return max, min, maxValIndex, minValIndex, sum, avg
}

func main() {
	s := []int{4, 66, 7, 2, 8, 55}
	max, min, maxIndex, minIndex, sum, avg := findMaxAndMin(s)

	fmt.Printf("最大值:%d\n", max)
	fmt.Printf("最小值:%d\n", min)
	fmt.Printf("最大值索引:%d\n", maxIndex)
	fmt.Printf("最小值索引:%d\n", minIndex)
	fmt.Printf("数组总和: %d\n", sum)
	fmt.Printf("数组平均值: %f\n", avg)
}
```
